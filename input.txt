wholenum mainCounter;
decimal result;

function divide(wholenum a, wholenum b) returns decimal {
    SAFE divisionBlock {
        result = a / b;
    }
    return result;
}

function main() returns nothing {

    mainCounter = 0;

    mainCounter from 0 to 10 in (mainCounter += 1) {

        when {
            is (mainCounter == 5) {
                print("Reached five");
            }
            is (mainCounter < 5) {
                print("Below five");
            }
            otherwise {
                print("Above five");
            }
        }

        bound(mainCounter) {
            in (1, 3) {
                print("Between one and three");
            }
            equals(5) {
                print("Exactly five");
            }
            above(8) {
                print("Greater than eight");
            }
        }
    }

    until(mainCounter > 15) {
        mainCounter = mainCounter + 1;
    }

    divide(10, 2);
}
